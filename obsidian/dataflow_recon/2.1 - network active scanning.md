#phase 

asnmap
	resolves ASN numbers into their full owned CIDR (Classless Inter-Domain Routing) ranges
		complements the manually compiled ip_ranges.txt that weren't obvious from BGP lookups
	`cat asns.txt | asnmap -silent >> asnmap_ranges.txt` 
	merge with ip_ranges.txt and deduplicate into a single authoritative CIDR list

naabu
	feed it the merged CIDR list and target web-relevant ports only
		OBS: going borad in port scanning wastes time and maximises noise
	
	cat asnmap_ranges.txt | naabu -p 80,443,8080,8443,8888,9000 -silent -o naabu_live_ports.txt

grab service banners
	echo <ASN_number> | asnmap -silent | naabu -nmap-cli 'nmap -sV'

-- input:
	asns.txt
	ip_ranges.txt

-- ouput:
	[[asnmap_ranges.txt]]
	[[naabu_live_ports.txt]]
